user
----
id integer pk
username varchar(32)
password_hash char(60)
email varchar(128) not null unique
avatar text
type enum('landlord','tenant','admin')
last_login_time timestamp with time zone not null
registered_at timestamp with time zone not null

property
-----
id integer pk
area varchar(16) not null
district varchar(16) not null
location varchar(16) not null
street varchar(16) not null
building varchar(16) not null
block varchar(4) 
floor varchar(2) 
room varchar(2) 
Rent varchar(16) not null
landlord_id fk>0- user.id not null
tenant_id fk>0- user.id
rental_start_at timestamp with time zone not null
rental_end_at timestamp with time zone
attachments_id fk>0- propertyAttachments.id

complaint
-----
id integer pk
description varchar(60)
type enum{'maintenance','notices','reimbursement','complaint'}
priority enum('high','medium','low')
status enum('resolved','pending','rejected')
content text
reason text
property_id fk>0- property.id not null
handled_by_id fk>0- user.id not null
handled_at timestamp with time zone
created_by_id fk>0- user.id not null
created_at timestamp with time zone
attachments_id fk>0- complaintAttachments.id

chatroom
----
id integer pk
creator_id fk>0- user.id not null
receiver_id fk>0- user.id not null
created_at timestamp with time zone not null

message
---
id integer pk
room_id fk>0- room.id not null
sender_id fk>0- user.id not null
content text
created_at timestamp with time zone not null

payment
---
id integer pk
property_id fk>0- property.id not null
billing_period enum('mon','feb','mar','apr','may','jun','jul','aug','sep','oct','nov','dec',)
payer_id fk>0- user.id not null
status enum('confirmed','pending') not null
created_by_id fk>0- user.id not null
created_at timestamp with time zone not null
confirmed_by_id fk>0- user.id not null
confirmed_at timestamp with time zone not null
attachments_id fk>0- paymentAttachments.id not null

complaintAttachments
---
id integer pk
attachments text not null
complaint_id fk>0- complaint.id not null

paymentAttachments
---
id integer pk
attachments text not null
payment_id fk>0- payment.id not null

propertyAttachments
---
id integer pk
attachments text not null
property_id fk>0- property.id not null

log
----
id integer pk
user_agent text
url text not null
method varchar(16) not null
