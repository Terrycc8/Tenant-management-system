user
----
id integer pk
username varchar(32)
password_hash char(60)
email varchar(128) not null
avatar text
last_online_time timestamp with time zone not null

property
-----
id integer pk
area varchar(16) not null
district varchar(16) not null
location varchar(16) not null
street varchar(16) not null
building varchar(16) not null
block varchar(4) 
floor varchar(2) 
room varchar(2) 
landlord_id fk>0- user.id not null
tenant_id fk>0- user.id
